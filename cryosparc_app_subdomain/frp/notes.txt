# frp_token needs to be 600 but readable by user apache in production
# file owned by you
install -o "$USER" -g "$USER" -m 600 /dev/null /path/to/.frp_token
printf '%s\n' 'YOUR_TOKEN' > /path/to/.frp_token

# allow apache to read the file
setfacl -m u:apache:r-- /path/to/.frp_token

# allow apache to traverse the directory path
setfacl -m u:apache:--x /path/to
setfacl -m u:apache:--x /path/to/parent

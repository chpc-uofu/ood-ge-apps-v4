Due to tricky permission issues with Apptainer direct build from Def
v4.7.1-cuda12 must to be built by Podman or Docker first and converted to sif 
See Dockerfile in v4.7.1 folder

v5.0.6 is OK with Apptainer direct build from Def

export LICENSE_ID=cd5d7832-ae84-11f0-b06a-df8cfaccd61d

# Pre-download installation packages
#export RELEASE_VERSION="v4.7.1-cuda12"
export RELEASE_VERSION="v5.0.6"
curl -L https://get.cryosparc.com/download/master-$RELEASE_VERSION/$LICENSE_ID -o ./cryosparc_master.tar.gz
curl -L https://get.cryosparc.com/download/worker-$RELEASE_VERSION/$LICENSE_ID -o ./cryosparc_worker.tar.gz


# Must build on a GPU node. 
cd v5
ml apptainer
unset APPTAINER_BINDPATH
export APPTAINER_CACHEDIR=$TMPDIR

# export LICENSE_ID=xxxxxx
export RELEASE_VERSION="v5.0.6"
export APPTAINERENV_LICENSE_ID=$LICENSE_ID
export APPTAINERENV_VERSION=$RELEASE_VERSION
apptainer build --fakeroot --nv -B $PWD:/mnt ./$VERSION.sif ./$VERSION.def 2>&1 | tee build.log
